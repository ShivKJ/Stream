  6# This module implements some of the useful implementations of "Collector"
  7# abstract class.
  8
  9# Currently there are following implementations:
 10# 1)  ToList: Collects stream elements into a "list"
 11# 2)  ToLinkedList: Collects stream elements into a "deque"
 12# 3)  ToSet: Collects stream elements into "set"
 13# 4)  CollectAndThen: firstly holds elements into a container using a downstream operation,
 14#                     and then applies a function on it.
 15# 5)  ToMap: Creates a dictionary from stream elements.
 16# 6)  Mapping: transforms stream elements and then collect then using downstream operation.
 17# 7)  MaxBy: Finds max element from stream (output will be of type "Optional")
 18# 8)  MinBy: Finds min element from stream (output will be of type "Optional")
 19# 9)  Joining: Joins string stream using separator "sep", "prefix" and "suffix".
 20# 10) Counting: finds number of element in stream.
 21# 11) Summing: sums elements of stream.
 22# 12) Averaging: finds average of elements of stream.
 23# 13) Reduce: Reduces stream elements using Binary function "bi_func". (output will be of type "Optional")
 24# 14) GroupingBy: groups stream elements into bucket (keys in dictionary are referred as buckets.).

 35class Collector(ABC):
 36    """
 37    Object of this class will be used in Stream.collect method.
 38    """
 39
 40    @abstractmethod
 41    def supply(self) -> 'Collector':
 42        """
 43        supplies a new Collector.
 44        :return:
 45        """
 46
 47    @abstractmethod
 48    def consume(self, e):
 49        """
 50        Defines how to process a data "e".
 51        :param e:
 52        """
 53
 54    @abstractmethod
 55    def finish(self):
 56        """
 57        defines what to return with "consumed" elements.
 58        :return:
 59        """

 63class DataHolder(Collector):
 64    """
 65    This class will be used to hold stream data into
 66    container such as 'list','set','deque' etc.
 67    """
 68
 69    def __init__(self, container_class):
 70        super().__init__()
 71
 72        self._data_holder = container_class()
 73        self._cls = container_class
 74
 75    def supply(self) -> Collector:
 76        return self.__class__()
 77
 78    @abstractmethod
 79    def consume(self, e): pass
 80
 81    def finish(self):
 82        return self._data_holder

127class CollectAndThen(Collector):
128    """
129    Collects elements using "downstream" and then applies the "then" function.
130
131    Stream([1,4,2,6,1,5,6]).collect(CollectAndThen(ToSet(),sum)) -> 18
132    Stream([1,4,2,6,1,5,6]).collect(CollectAndThen(ToList(),sum)) -> 25
133
134    """
135
136    def __init__(self, downstream: Collector, then):
137        self._downstream = downstream
138        self._then = then
139
140    def supply(self) -> Collector:
141        return CollectAndThen(self._downstream.supply(), self._then)
142
143    def consume(self, e):
144        self._downstream.consume(e)
145
146    def finish(self):
147        return self._then(self._downstream.finish())

159class ToMap(Collector):
160    """
161    Used to create map from data points.
162
163    Stream(range(5)).collect(ToMap(lambda x:x,lambda x:x**2))
164    -> {0: 0, 1: 1, 2: 4, 3: 9, 4: 16}
165
166    Stream([1,4,2,6,1,5,6]).collect(ToMap(lambda x:x,lambda x:x**2,
167                                          merger_on_conflict=lambda o,n:o))
168    -> {1: 1, 4: 16, 2: 4, 6: 36, 5: 25}
169
170    """
171
172    def __init__(self, key_mapper: Function, value_mapper: Function = identity,
173                 merger_on_conflict: BiFunction = None):
174        super().__init__()
175
176        self._key_mapper = key_mapper
177        self._value_mapper = value_mapper
178        self._merger_on_conflict = merger_on_conflict
179
180        self._data_holder = {}
181
182    def supply(self) -> Collector:
183        return ToMap(self._key_mapper, self._value_mapper, self._merger_on_conflict)
184
185    def consume(self, e):
186        """
187        creates an Entry in with key using function '_key_mapper' and value
188        using function '_value_mapper'. Note that these key-value pairs are stored
189        in a dictionary so key made must to hashable.
190
191        In case, key has already been created with earlier element, a ValueError
192        will be thrown if "_merge_on_conflict" is not set, else this BiFunction
193        will be used to resolve the conflict.
194
195        Example:
196            from streamAPI.stream import *
197            from streamAPI.stream.TO import *
198
199            Stream([1,2,3,4]).collect(ToMap(lambda x:x,lambda x:x**2))
200            -> {1: 1, 2: 4, 3: 9, 4: 16}
201
202            Stream([1,2,1,4]).collect(ToMap(lambda x:x,lambda x:x**2))
203            will throw ValueError as element '1' is present multiple times.
204
205
206            Stream([1,2,1,4]).collect(ToMap(lambda x:x,lambda x:x**2,lambda o,n:o))
207            -> {1: 1, 2: 4, 4: 16} # in case of conflict, we are using old value.
208
209
210        :param e:
211        :return:
212        """
213
214        bkt = self._key_mapper(e)
215
216        if bkt in self._data_holder:
217            if self._merger_on_conflict is None:
218                raise ValueError(f'k : {bkt} is already present.')
219
220            old_v = self._data_holder[bkt]
221
222            self._data_holder[bkt] = self._merger_on_conflict(old_v, self._value_mapper(e))
223        else:
224            self._data_holder[bkt] = self._value_mapper(e)
225
226    def finish(self) -> dict:
227        return self._data_holder

230class Mapping(Collector):
231    """
232    Maps elements using "func" function first, then collects transformed
233    elements according to "downstream".
234
235    Stream(range(5)).collect(Mapping(lambda x:x**2,Summing()))
236    -> 30 # sum of squares
237    """
238
239    def __init__(self, func: Function, downstream: Collector = None):
240        super().__init__()
241
242        self._func = func
243        self._downstream = downstream or ToList()
244
245    def supply(self) -> Collector:
246        return Mapping(self._func, self._downstream.supply())
247
248    def consume(self, e):
249        self._downstream.consume(self._func(e))
250
251    def finish(self):
252        return self._downstream.finish()

255class MaxBy(Collector):
256    """
257    Finds max element using comparator "comp"
258    Stream([1,4,2,6,1,5,6]).collect(MaxBy()) -> Optional[6]
259    """
260
261    def __init__(self, comp: BiFunction[X, X, int] = default_comp):
262        super().__init__()
263
264        self._max = NIL
265        self._comp = comp
266
267    def supply(self) -> Collector:
268        return MaxBy(self._comp)
269
270    def consume(self, e):
271        if self._max is NIL:
272            self._max = e
273        elif self._comp(e, self._max) > 0:
274            self._max = e
275
276    def finish(self) -> Optional:
277        return create_optional(self._max)

392class Reduce(Collector):
393    """
394    Reduces stream data.
395
396    import operator as op
397    Stream(range(1,5)).collect(Reduce(bi_func=op.mul)) -> Optional[24]
398    """
399
400    def __init__(self, o=NIL, *, bi_func: BiFunction):
401        """
402        Reduces Stream data.
403
404        :param o: initial point, defaults to NIL
405        :param bi_func: reducing function
406
407        """
408
409        super().__init__()
410
411        self._o = o
412        self._bi_func = bi_func
413
414        self._data_holder = o
415
416    def supply(self) -> Collector:
417        return Reduce(self._o, bi_func=self._bi_func)
418
419    def consume(self, e):
420        if self._data_holder is NIL:
421            self._data_holder = e
422        else:
423            self._data_holder = self._bi_func(self._data_holder, e)
424
425    def finish(self) -> Optional:
426        return create_optional(self._data_holder)

429class GroupingBy(Collector):
430    """
431    Groups stream elements.
432
433    Stream(['AB','BD','AC','DE','BD','BW','AB']).collect(GroupingBy(lambda x:x[0],
434                                                                    GroupingBy(lambda x:x[1],
435                                                                                Counting())))
436    -> {'A': {'B': 2, 'C': 1}, 'B': {'D': 2, 'W': 1}, 'D': {'E': 1}}
437    """
438
439    def __init__(self, group_by: Function, downstream: Collector = None):
440        super().__init__()
441
442        self._group_by = group_by
443        self._downstream = downstream or ToList()
444        self._bucket: DefaultDict[Any, Collector] = defaultdict(self._downstream.supply)
445
446    def supply(self) -> Collector:
447        return GroupingBy(self._group_by, self._downstream.supply())
448
449    def consume(self, e):
450        bkt = self._group_by(e)
451        self._bucket[bkt].consume(e)
452
453    def finish(self) -> dict:
454        return {k: v.finish() for k, v in self._bucket.items()}
