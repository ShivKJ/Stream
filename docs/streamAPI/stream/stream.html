  19class Stream(Closable, Generic[X]):
  20    """
  21    This class can be used to create pipeline operation on given
  22    iterable object.
  23
  24    There are two type of operation that can be performed on Stream:
  25    1) Intermediate (i.e. map, flat_map, filter, peek, distinct, sort, batch, cycle, take_while, drop_while etc.)
  26    2) Terminal (min, max, as_seq, group_by, all, any, none_match, for_each etc.)
  27
  28    Until terminal operation is called, no execution takes place.
  29    Some of the examples are given below.

 337    @staticmethod
 338    def _consumer_wrapper(consumer: Consumer[X], n: int = 1):
 339        """
 340        Creates a wrapper around consumer.
 341
 342        :param consumer:
 343        :param n: a natural number
 344        :return: a decorator which enables iterable to be consumed after each 'n'
 345        """
 346
 347        @wraps(consumer)
 348        def func(itr: Iterable[X]) -> Iterable[X]:
 349            one_to_n = cycle(range(1, n + 1))  # cycling numbers from 1 up to n
 350
 351            for idx, g in zip(one_to_n, itr):
 352                if idx == n:
 353                    consumer(g)
 354
 355                yield g
 356
 357        return func

 455    @check_pipeline
 456    def zip(self, *itr: Iterable[Y], after=True) -> 'Stream[Tuple]':
 457        """
 458        zips stream with another Iterable object.
 459
 460        We can specify whether to zip iterable after the stream of before
 461        the stream by using "after".
 462
 463        zip operation will produce a stream which will be exhausted if either
 464        itr has been exhausted or underlying stream is exhausted.
 465
 466        Example:
 467            Stream(range(100, 100000)).zip(range(5)).collect(ToList())
 468            -> [(100, 0), (101, 1), (102, 2), (103, 3), (104, 4)]
 469
 470            Stream(range(5)).zip(range(100, 100000)).collect(ToList())
 471            -> [(0, 100), (1, 101), (2, 102), (3, 103), (4, 104)]
 472
 473            Stream(range(20, 30)).zip(range(5),after=False).collect(ToList())
 474            # data from range(5) will be used as first entry of tuple created by zipping
 475            # stream with iterable
 476            -> [(0, 20), (1, 21), (2, 22), (3, 23), (4, 24)]
 477
 478        :param itr:
 479        :param after
 480        :return: Stream itself
 481        """
 482
 483        if after:
 484            self._pointer = zip(self._pointer, *itr)
 485        else:
 486            self._pointer = zip(*itr, self._pointer)
 487
 488        return self

 491    def zip_longest(self, *itr: Iterable[Y], after=True, fillvalue=None) -> 'Stream[Tuple]':
 492        """

 494        zip_longest allow stream generator even though smaller iterable has been exhausted.

 601    @check_pipeline
 602    def window_function(self, func, n: Union[int, None]) -> 'Stream[X]':
 603        """
 604        If "n" is not None (then it has to be an integer) then invoking function
 605        "func" on 'tuple' of "n" elements of stream. 'tuple' is made using past
 606        n-1 elements and 1 current element.
 607
 608        If "n" is None, then all past values up to current value, held by a 'list',
 609        will be sent to function "func". Note that data type 'list', a mutable object,
 610        is chosen to make process more memory efficient. Any change made in the 'list'
 611        inside "func" will be visible to in future call to "func".(a 'list' is
 612        mutated via 'append','clean','extend','insert','pop','remove', 'reverse',
 613        'sort' methods.) Each call to "func" will be sent the same list appended
 614        with current element.
 615
 616        Example1: Moving average for window size 3
 617            def mean(l): return sum(l)/len(l)
 618
 619            Stream([1,6,2,7,3]).window_function(mean , 3).collect(ToList())
 620            -> [3.0, 5.0, 4.0]
 621
 622        Example2: Averaging all past values:
 623            def mean(l): return sum(l)/len(l)
 624
 625            Stream(range(1,5)).window_function(mean , None).collect(ToList())
 626            -> [1.0, 1.5, 2.0, 2.5]
 627
 628        :param func: if "n" is not None then, takes input, at any instant,
 629                     as a 'tuple' having past "n-1" elements appended with
 630                     current element.
 631
 632                     If "n" is None, the all past values in a list appended
 633                     with current element.
 634        :param n: natural number or None
 635        :return: Stream itself
 636        """
 637
 638        if n is None:
 639            itr = Stream._all_past_values(self._pointer)
 640        else:
 641            itr = Stream._fetch_next(self._pointer, n)
 642
 643        self._pointer = map(func, itr)
 644        return self

 664    @staticmethod
 665    def _fetch_next(itr: Iterable[X], n: int) -> Iterable[Tuple[X, ...]]:
 666        """
 667        Creates generator. Each element is tuple of size "n".
 668
 669        While iterating, first element will be first "n" elements of
 670        Stream, next element will be old "n-1" elements appended with
 671        next element of Stream and so on.
 672
 673        Note that if in first call to this generator, stream has less
 674        than "n" element then ValueError will be thrown.
 675
 676        :param itr:
 677        :param n: a natural number
 678        :return:
 679        """
 680
 681        if n < 1:
 682            raise ValueError("'n' must be natural number")

 684        chunk = get_chunk(itr, n)
 685
 686        if len(chunk) != n:
 687            # first chunk size must be "n".
 688            raise ValueError("Stream has less than '{}' elements ".format(n))
 689
 690        yield chunk
 691
 692        for e in itr:
 693            chunk = chunk[1:] + (e,)
 694            yield chunk

 350            one_to_n = cycle(range(1, n + 1))  # cycling numbers from 1 up to n

 354                    consumer(g)

 458        """
 459        zips stream with another Iterable object.
 460

 462        the stream by using "after".

 465        itr has been exhausted or underlying stream is exhausted.

 602    @check_pipeline
 603    def window_function(self, func, n: Union[int, None]) -> 'Stream[X]':
 604        """
 605        If "n" is not None (then it has to be an integer) then invoking function
 606        "func" on 'tuple' of "n" elements of stream. 'tuple' is made using past
 607        n-1 elements and 1 current element.
 608
 609        If "n" is None, then all past values up to current value, held by a 'list',
 610        will be sent to function "func". Note that data type 'list', a mutable object,
 611        is chosen to make process more memory efficient. Any change made in the 'list'
 612        inside "func" will be visible to in future call to "func".(a 'list' is
 613        mutated via 'append','clean','extend','insert','pop','remove', 'reverse',
 614        'sort' methods.) Each call to "func" will be sent the same list appended
 615        with current element.
 616
 617        Example1: Moving average for window size 3
 618            def mean(l): return sum(l)/len(l)
 619
 620            Stream([1,6,2,7,3]).window_function(mean , 3).collect(ToList())
 621            -> [3.0, 5.0, 4.0]
 622
 623        Example2: Averaging all past values:
 624            def mean(l): return sum(l)/len(l)
 625
 626            Stream(range(1,5)).window_function(mean , None).collect(ToList())
 627            -> [1.0, 1.5, 2.0, 2.5]
 628
 629        :param func: if "n" is not None then, takes input, at any instant,
 630                     as a 'tuple' having past "n-1" elements appended with
 631                     current element.
 632
 633                     If "n" is None, the all past values in a list appended
 634                     with current element.
 635        :param n: natural number or None
 636        :return: Stream itself
 637        """
 638
 639        if n is None:
 640            itr = Stream._all_past_values(self._pointer)
 641        else:
 642            itr = Stream._fetch_next(self._pointer, n)
 643
 644        self._pointer = map(func, itr)
 645        return self

 647    @staticmethod
 648    def _all_past_values(itr: Iterable[X]) -> Iterable[list]:
 649        """
 650        Creates a generator.
 651
 652        Generator always returns same list but each time an element from 'itr'
 653        is appended to the list.
 654
 655        :param itr:
 656        :return:

 659        data_holder = []

 661        for e in itr:

 665    @staticmethod
 666    def _fetch_next(itr: Iterable[X], n: int) -> Iterable[Tuple[X, ...]]:
 667        """
 668        Creates generator. Each element is tuple of size "n".
 669
 670        While iterating, first element will be first "n" elements of
 671        Stream, next element will be old "n-1" elements appended with
 672        next element of Stream and so on.
 673
 674        Note that if in first call to this generator, stream has less
 675        than "n" element then ValueError will be thrown.
 676
 677        :param itr:
 678        :param n: a natural number
 679        :return:
 680        """
 681
 682        if n < 1:
 683            raise ValueError("'n' must be natural number")
 684
 685        chunk = get_chunk(itr, n)
 686
 687        if len(chunk) != n:
 688            # first chunk size must be "n".
 689            raise ValueError("Stream has less than '{}' elements ".format(n))
 690
 691        yield chunk
 692
 693        for e in itr:
 694            chunk = chunk[1:] + (e,)
 695            yield chunk

This class can be used to create pipeline operation on given iterable object.

There are two type of operation that can be performed on Stream: 1) Intermediate (i.e. map, flat_map, filter, peek, distinct, sort, batch, cycle, take_while, drop_while etc.) 2) Terminal (min, max, as_seq, group_by, all, any, none_match, for_each etc.)

Until terminal operation is called, no execution takes place. Some of the examples are given below.

        return '[name='+self.name+',age='+str(self.age)+',sex='+('Male' if self.sex else 'Female')+']'
