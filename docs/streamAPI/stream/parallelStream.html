 21    def __init__(self, data: Iterable[T],
 22                 worker: int = None,
 23                 multiprocessing: bool = True):
 24        """
 25        Initialises executor which will be used for processing stream elements.
 26
 27        If "worker" is None, then in case of multiprocessing number of cpu in
 28        the system is used and in case of multiThreading (i.e multiprocessing = False)
 29        "worker" is 5 * cpu_count().
 30
 31        :param data:
 32        :param worker: number of worker
 33        :param multiprocessing: it True then multiprocessing is used else multiThreading.
 34        """
 35
 36        super().__init__(data)
 37
 38        self._registered_jobs: Deque[Future] = deque()
 39
 40        if worker is None:
 41            worker = Exec._default_worker(multiprocessing)
 42
 43        self._exec: Executor = (PPE if multiprocessing else TPE)(max_workers=worker)
 44        self._worker = worker

 47    def _default_worker(multiprocessing: bool):
 48        """
 49        returns number of worker for concurrent processing of stream elements.
 50
 51        In case of multiprocessing, it is number of cpu_count() and for
 52        multiThreading it is 5 times cpu_count().
 53
 54        :param multiprocessing:
 55        :return:
 56        """

 58        worker = cpu_count() or 1

 62    def _parallel_processor(self: 'ParallelStream[T]', func, timeout=None, batch_size=None) -> Stream[T]:
 63        """
 64        processes data points concurrently. Elements are processed in batches of size "batch_size".
 65
 66        :param self:
 67        :param func:
 68        :param timeout: time to wait for task to be done, if None then there is no
 69                        limit on execution time.
 70        :param batch_size: If it is None then number of worker is used.
 71        :return:
 72        """
 73
 74        batch_size = batch_size or self._worker
 75
 76        assert batch_size > 0, 'Batch size must be positive.'
 77
 78        stream = (Stream(iter(self._pointer))
 79                  .map(partial(self._submit_job, func))
 80                  .peek(self._registered_jobs.append)
 81                  .batch(batch_size)
 82                  .map(as_completed)
 83                  .flat_map())
 84
 85        if timeout is not None:
 86            result_extractor = partial(Future.result, timeout=timeout)
 87        else:
 88            result_extractor = Future.result
 89
 90        return stream.map(result_extractor)

 93        """
 94        Submits job to executor.
 95
 96        :param func:
 97        :param g:
 98        :return:
 99        """
100
101        return self._exec.submit(func, g)

157    @check_pipeline
158    def batch_processor(self, func: Function[T, X], dispatch_size: int, timeout=None):
159        """
160        This method is advised to be invoked when using MultiProcessing.
161
162        Usually dispatching single unit of work (that is applying function on
163        a element) is less costly. So it is better to send a batch of elements to
164        a worker. Here "dispatch_size" is size of dispatch i.e. this many number
165        of stream elements will be sent to each processor(worker) in one go.
166
167        :param func:
168        :param dispatch_size: number of stream elements to be given to each worker
169                              in one go.
170        :param timeout: time to wait for task to be done, if None then there is no
171                        limit on execution time.
172        :return:
173        """
174
175        return (self.batch(dispatch_size)
176                .map_concurrent(partial(ParallelStream._batch_process, func), timeout=timeout)
177                .flat_map())

185        :param timeout: time to wait for task to be done, if None then there is no

This class can be used to create pipeline operation on given iterable object.

There are two type of operation that can be performed on Stream: 1) Intermediate (i.e. map, flat_map, filter, peek, distinct, sort, batch, cycle, take_while, drop_while etc.) 2) Terminal (min, max, as_seq, group_by, all, any, none_match, for_each etc.)

Until terminal operation is called, no execution takes place. Some of the examples are given below.

        self.name = name
